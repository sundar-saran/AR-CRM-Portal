ALTER PROCEDURE [dbo].[sp_Sundar_CRM_Project]
    @Action NVARCHAR(50),
    @UserId INT = NULL,
    @FullName NVARCHAR(100) = NULL,
    @Mobile NVARCHAR(15) = NULL,
    @Email NVARCHAR(100) = NULL,
    @City NVARCHAR(50) = NULL,
    @Password NVARCHAR(255) = NULL,
    @HashedPassword NVARCHAR(255) = NULL,
    @Role_Id INT = NULL,
    @CreatedOn INT = NULL,
    @ProjectId INT = NULL,
    @ProjectName NVARCHAR(100) = NULL,
    @Description NVARCHAR(MAX) = NULL,
    @IsActive BIT = NULL,
    @UserProjectId INT = NULL,
    @Status NVARCHAR(20) = NULL,
    @NewRoleId INT = NULL,
    @UpdaterId INT = NULL,
    @SearchEmail NVARCHAR(100) = NULL,
    @SearchMobile NVARCHAR(15) = NULL,
    @TargetUserId INT = NULL,
    @ColumnName NVARCHAR(128) = NULL,
    @DataType NVARCHAR(50) = NULL,
    @LeadRowData NVARCHAR(MAX) = NULL,
    @Columns NVARCHAR(MAX) = NULL,
    @Values NVARCHAR(MAX) = NULL,
    @IsRequired BIT = 0,
    @LeadId INT = NULL,
    @ApprovedByUserId INT = NULL,
    @MenuURL NVARCHAR(200) = NULL,
    @Menu_ID INT = NULL,
    @AccessId INT = NULL,
    @MenuId NVARCHAR(500) = NULL,
    @MenuIds NVARCHAR(500) = NULL
	AS
	BEGIN
		SET NOCOUNT ON;
		DECLARE @NewUserId INT, @UserMenuIds NVARCHAR(50);

		-- User Management Actions
		IF @Action IN ('RegisterUser', 'AddNewUser', 'ValidateUser')
			SET @HashedPassword = CONVERT(NVARCHAR(255), HASHBYTES('SHA2_256', @Password), 2);

		IF @Action IN ('RegisterUser', 'UpdateUserRole')
			SET @MenuIds = CASE 
							WHEN @Role_Id = 2 OR @NewRoleId = 2 THEN '1,2,3,5,8,9,10,12' 
							WHEN @Role_Id = 1 OR @NewRoleId = 1 THEN '1,4,6,7,11' 
							ELSE '' 
						  END;

		-- User Registration Action
		IF @Action = 'RegisterUser'
		BEGIN
			-- Check if the user already exists by email or mobile
			IF EXISTS (SELECT 1 FROM Sundar_tbl_UserDetails WHERE (Email = @Email OR Mobile = @Mobile) AND is_Delete = 0)
				SELECT 'User with this email or mobile already exists.' AS Message, 0 AS Flag;
			ELSE
			BEGIN
				-- Insert new user and get the newly inserted User_ID
				INSERT INTO Sundar_tbl_UserDetails (FullName, Mobile, Email, City, Password, Created_On)
				OUTPUT INSERTED.User_ID 
				VALUES (@FullName, @Mobile, @Email, @City, @HashedPassword, ISNULL(@CreatedOn, SCOPE_IDENTITY()));

				SET @NewUserId = SCOPE_IDENTITY();

				-- Insert user access data (assign roles and menu items)
				INSERT INTO Sundar_tbl_Access (User_ID, Role_ID, Menu_ID, Created_On) 
				VALUES (@NewUserId, @Role_Id, @MenuIds, ISNULL(@CreatedOn, @NewUserId));

				SELECT 'User registered successfully' AS Message, 1 AS Flag, @NewUserId AS UserId;
			END
		END

		-- Combined User Data Retrieval Action
		ELSE IF @Action = 'GetUserData'
		BEGIN
			SELECT 
				u.User_ID, u.FullName, u.Mobile, u.Email, u.City, u.is_Active, u.Created_Date,
				a.Role_ID, r.RoleName,
				CASE 
					WHEN @UserId IS NOT NULL AND u.User_ID = @UserId THEN 'SingleUser'
					WHEN @Email IS NOT NULL AND u.Email = @Email THEN 'SingleUser'
					ELSE 'AllUsers'
				END AS ResultType
			FROM Sundar_tbl_UserDetails u
			LEFT JOIN Sundar_tbl_Access a ON u.User_ID = a.User_ID AND a.is_Delete = 0
			LEFT JOIN Sundar_tbl_Role r ON a.Role_ID = r.Role_ID
			WHERE u.is_Delete = 0
			  AND (
				(@UserId IS NOT NULL AND u.User_ID = @UserId) OR
				(@Email IS NOT NULL AND u.Email = @Email) OR
				(@UserId IS NULL AND @Email IS NULL)
			  );
		END

		-- Combined User Access Data Action
		ELSE IF @Action = 'GetAccessData'
		BEGIN
			SELECT 
				u.User_ID, u.FullName, u.Email, u.Mobile, a.Role_ID, r.RoleName,
				CASE 
					WHEN @UserId IS NOT NULL THEN 'UserAccess'
					ELSE 'AllAccessData'
				END AS ResultType
			FROM Sundar_tbl_UserDetails u
			INNER JOIN Sundar_tbl_Access a ON u.User_ID = a.User_ID
			INNER JOIN Sundar_tbl_Role r ON a.Role_ID = r.Role_ID
			WHERE u.is_Delete = 0
			  AND (@UserId IS NULL OR u.User_ID = @UserId);
		END

		-- Validate User Action
		IF @Action = 'ValidateUser'
	BEGIN
		SET @HashedPassword = CONVERT(NVARCHAR(255), HASHBYTES('SHA2_256', @Password), 2);

		SELECT ud.User_ID, a.Role_ID 
		FROM Sundar_tbl_UserDetails ud
		JOIN Sundar_tbl_Access a ON ud.User_ID = a.User_ID
		WHERE ud.Email = @Email 
		AND ud.Password = @HashedPassword
		AND ud.is_Delete = 0
		AND ud.is_Active = 1
		AND a.Role_ID IS NOT NULL;
	END


		-- Combined User Update Action
		ELSE IF @Action = 'UpdateUser'
		BEGIN
			UPDATE Sundar_tbl_UserDetails 
			SET 
				FullName = ISNULL(@FullName, FullName),
				Mobile = ISNULL(@Mobile, Mobile),
				City = ISNULL(@City, City),
				is_Active = CASE WHEN @IsActive IS NOT NULL THEN @IsActive ELSE is_Active END,
				Update_Date = GETDATE(), 
				Update_on = ISNULL(@UpdaterId, Update_on)
			WHERE User_ID = @UserId AND is_Delete = 0;
        
			SELECT @@ROWCOUNT as RowsAffected;
		END

    ELSE IF @Action = 'UpdateUserRole'
    BEGIN
        UPDATE Sundar_tbl_Access 
        SET Role_ID = @NewRoleId, Menu_ID = @MenuIds, Update_Date = GETDATE(), Update_on = @UpdaterId
        WHERE User_ID = @UserId AND is_Delete = 0;
        SELECT @@ROWCOUNT;
    END

    ELSE IF @Action = 'AddNewUser'
	BEGIN
		BEGIN TRY
			BEGIN TRANSACTION;

			-- Check if user already exists
			IF EXISTS (SELECT 1 FROM Sundar_tbl_UserDetails WHERE (Email = @Email OR Mobile = @Mobile) AND is_Delete = 0)
			BEGIN
				SELECT 0 AS Success;
				RETURN;
			END

			SET @HashedPassword = CONVERT(NVARCHAR(255), HASHBYTES('SHA2_256', @Password), 2);

			INSERT INTO Sundar_tbl_UserDetails (FullName, Email, Mobile, City, Password, is_Delete, is_Active, Created_Date, Created_On) 
			VALUES (@FullName, @Email, @Mobile, @City, @HashedPassword, 0, 1, GETDATE(), 1);
        
			SET @NewUserId = SCOPE_IDENTITY();
        
			INSERT INTO Sundar_tbl_Access (User_ID, Role_ID, Menu_ID, is_Delete, is_Active, Created_Date, Created_On) 
			VALUES (@NewUserId, 1, '1,4,6,7,11', 0, 1, GETDATE(), 1);
        
			COMMIT TRANSACTION;
			SELECT 1 AS Success;
		END TRY
		BEGIN CATCH
			ROLLBACK TRANSACTION;
			SELECT 0 AS Success;
		END CATCH
	END

    -- Menu and Access Management
    ELSE IF @Action = 'GetUserMenus'
    BEGIN
        SELECT @UserMenuIds = Menu_ID FROM Sundar_tbl_Access WHERE User_ID = @UserId;
        SELECT Menu_ID, MenuName, MenuURL FROM Sundar_tbl_Menu 
        WHERE CHARINDEX(',' + CAST(Menu_ID AS NVARCHAR(10)) + ',', ',' + @UserMenuIds + ',') > 0;
    END

    ELSE IF @Action = 'GetRoles'
    BEGIN
        SELECT Role_ID, RoleName FROM Sundar_tbl_Role ORDER BY RoleName;
    END

    -- Combined Project Management Action
    ELSE IF @Action = 'ManageProject'
    BEGIN
        IF @ProjectId IS NULL
        BEGIN
            INSERT INTO Sundar_tbl_Project (ProjectName, Description, is_Active, Created_Date) 
            VALUES (@ProjectName, @Description, 1, GETDATE());
            SELECT SCOPE_IDENTITY() as NewProjectId, @@ROWCOUNT as RowsAffected;
        END
        ELSE -- Update existing project
        BEGIN
            UPDATE Sundar_tbl_Project 
            SET ProjectName = @ProjectName, 
                Description = @Description,
                is_Active = ISNULL(@IsActive, is_Active)
            WHERE Project_ID = @ProjectId;
            SELECT @ProjectId as ProjectId, @@ROWCOUNT as RowsAffected;
        END
    END

    -- Combined Project Data Retrieval Action
    ELSE IF @Action = 'GetProjectData'
    BEGIN
        SELECT 
            Project_ID, ProjectName, Description, is_Active, Created_Date,
            CASE 
                WHEN @ProjectId IS NOT NULL THEN 'SingleProject'
                ELSE 'AllProjects'
            END as ResultType
        FROM Sundar_tbl_Project 
        WHERE (@ProjectId IS NULL OR Project_ID = @ProjectId)
          AND (@IsActive IS NULL OR is_Active = @IsActive)
        ORDER BY 
            CASE WHEN @ProjectId IS NULL THEN Created_Date END DESC, 
            ProjectName;
    END

    -- Combined User Projects Action
	ELSE IF @Action = 'ManageUserProjects'
	BEGIN
		IF @Status = 'Apply' -- Apply to project
		BEGIN
			IF NOT EXISTS (SELECT 1 FROM Sundar_tbl_UserProjects WHERE User_ID = @UserId AND Project_ID = @ProjectId)
				INSERT INTO Sundar_tbl_UserProjects (User_ID, Project_ID, Status) VALUES (@UserId, @ProjectId, 'Pending');
			SELECT @@ROWCOUNT as RowsAffected;
		END
		ELSE IF @Status IS NOT NULL AND @UserProjectId IS NOT NULL -- Update status
		BEGIN
			UPDATE Sundar_tbl_UserProjects 
			SET Status = @Status, Update_Date = GETDATE() 
			WHERE UserProject_ID = @UserProjectId;
			SELECT @@ROWCOUNT as RowsAffected;
		END
		ELSE IF @UserId IS NOT NULL AND @Status IS NOT NULL -- Get user projects by status (ADD THIS CONDITION)
		BEGIN
			SELECT 
				up.UserProject_ID, up.Status, up.User_ID, p.Project_ID, p.ProjectName, p.Description,
				u.FullName, u.Email, u.Mobile,
				'UserProjectsByStatus' as ResultType
			FROM Sundar_tbl_UserProjects up
			JOIN Sundar_tbl_Project p ON up.Project_ID = p.Project_ID
			JOIN Sundar_tbl_UserDetails u ON up.User_ID = u.User_ID 
			WHERE u.is_Delete = 0
			  AND up.User_ID = @UserId
			  AND up.Status = @Status;
		END
		ELSE IF @UserId IS NOT NULL -- Get all user projects
		BEGIN
			SELECT 
				up.UserProject_ID, up.Status, up.User_ID, p.Project_ID, p.ProjectName, p.Description,
				u.FullName, u.Email, u.Mobile,
				'UserProjects' as ResultType
			FROM Sundar_tbl_UserProjects up
			JOIN Sundar_tbl_Project p ON up.Project_ID = p.Project_ID
			JOIN Sundar_tbl_UserDetails u ON up.User_ID = u.User_ID 
			WHERE u.is_Delete = 0
			  AND up.User_ID = @UserId;
		END

		ELSE -- Get all applications
		BEGIN
			SELECT 
				up.UserProject_ID, up.Status, up.User_ID, p.Project_ID, p.ProjectName, p.Description,
				u.FullName, u.Email, u.Mobile,
				'AllApplications' as ResultType
			FROM Sundar_tbl_UserProjects up
			JOIN Sundar_tbl_Project p ON up.Project_ID = p.Project_ID
			JOIN Sundar_tbl_UserDetails u ON up.User_ID = u.User_ID 
			WHERE u.is_Delete = 0;
		END
	END
	IF @Action = 'GetApprovedProjectsByUser'
		BEGIN
			SELECT 
			p.Project_ID,
			p.ProjectName, 
			p.Description,
			up.Status
		FROM Sundar_tbl_UserProjects up
		JOIN Sundar_tbl_Project p ON up.Project_ID = p.Project_ID
		WHERE up.User_ID = @UserId AND up.Status = 'Assigned'
		END
    -- Lead Management Actions
    ELSE IF @Action = 'GetColumns'
	BEGIN
		SELECT COLUMN_NAME, DATA_TYPE
		FROM INFORMATION_SCHEMA.COLUMNS
		WHERE TABLE_NAME = 'Sundar_tbl_Lead'
		  AND COLUMN_NAME NOT IN ('Id', 'CreatedByUserId', 'CreatedDate')
		ORDER BY ORDINAL_POSITION;
	END

	ELSE IF @Action = 'AddColumn'
	BEGIN
		IF EXISTS (SELECT 1 FROM INFORMATION_SCHEMA.COLUMNS WHERE TABLE_NAME = 'Sundar_tbl_Lead' AND COLUMN_NAME = @ColumnName)
		BEGIN
			SELECT 'Column exists' AS Message, 0 AS Flag;
		END
		ELSE
		BEGIN
			BEGIN TRY
				DECLARE @sqlAdd NVARCHAR(MAX) = N'ALTER TABLE dbo.Sundar_tbl_Lead ADD ' + QUOTENAME(@ColumnName) + ' ' + @DataType + ' NULL';
				EXEC sp_executesql @sqlAdd;
				SELECT 'Column added successfully' AS Message, 1 AS Flag;
			END TRY
			BEGIN CATCH
				SELECT 'Error adding column: ' + ERROR_MESSAGE() AS Message, 0 AS Flag;
			END CATCH
		END
	END

	ELSE IF @Action = 'DeleteColumn'
	BEGIN
		IF NOT EXISTS (SELECT 1 FROM INFORMATION_SCHEMA.COLUMNS WHERE TABLE_NAME = 'Sundar_tbl_Lead' AND COLUMN_NAME = @ColumnName)
		BEGIN
			SELECT 'Column does not exist' AS Message, 0 AS Flag;
		END
		ELSE
		BEGIN
			DECLARE @sqlDrop NVARCHAR(MAX) = N'ALTER TABLE Sundar_tbl_Lead DROP COLUMN ' + QUOTENAME(@ColumnName);
			EXEC sp_executesql @sqlDrop;
			SELECT 'Column deleted' AS Message, 1 AS Flag;
		END
	END

	ELSE IF @Action = 'InsertLead'
	BEGIN
		IF @UserId IS NULL OR @Columns IS NULL OR @Values IS NULL
		BEGIN
			SELECT 'Invalid input data' AS Message, 0 AS Flag;
			RETURN;
		END

		DECLARE @sqlInsert NVARCHAR(MAX);

		SET @sqlInsert = '
			INSERT INTO dbo.Sundar_tbl_Lead (CreatedByUserId, ' + @Columns + ', CreatedDate)
			VALUES (' + CAST(@UserId AS NVARCHAR) + ', ' + @Values + ', GETDATE())
		';

		BEGIN TRY
			EXEC sp_executesql @sqlInsert;
			SELECT 'Lead inserted successfully' AS Message, 1 AS Flag;
		END TRY
		BEGIN CATCH
			SELECT 'Insert failed: ' + ERROR_MESSAGE() AS Message, 0 AS Flag;
		END CATCH
	END

	ELSE IF @Action = 'GetLeads'
	BEGIN
		DECLARE @colsSelect NVARCHAR(MAX);

		-- Use XML PATH method for string aggregation (compatible with older SQL versions)
		SELECT @colsSelect = COALESCE(@colsSelect + ', ', '') + QUOTENAME(COLUMN_NAME)
		FROM INFORMATION_SCHEMA.COLUMNS
		WHERE TABLE_NAME = 'Sundar_tbl_Lead'
		  AND COLUMN_NAME NOT IN ('Id', 'CreatedByUserId', 'CreatedDate');

		SET @colsSelect = ISNULL(@colsSelect, '');

		DECLARE @sqlSelect NVARCHAR(MAX) = 
			N'SELECT Id, CreatedByUserId, CreatedDate' 
			+ CASE WHEN @colsSelect <> '' THEN ', ' + @colsSelect ELSE '' END
			+ N' FROM dbo.Sundar_tbl_Lead'
			+ CASE WHEN @UserId IS NOT NULL THEN N' WHERE CreatedByUserId = ' + CAST(@UserId AS NVARCHAR(10)) ELSE '' END;

		EXEC sp_executesql @sqlSelect;
	END

	ELSE IF @Action = 'GetAllLeadApplications'
	BEGIN
		SELECT 
			l.Id,
			l.CreatedByUserId,
			u.FullName as UserName,
			u.Email as UserEmail,
			l.CreatedDate
		FROM Sundar_tbl_Lead l
		INNER JOIN Sundar_tbl_UserDetails u ON l.CreatedByUserId = u.User_ID
		ORDER BY l.CreatedDate DESC;
	END

	-- Menu Management Actions
	ELSE IF @Action = 'AddMenu'
	BEGIN
		IF @Role_Id != 2
		BEGIN
			SELECT 'Access Denied: Admin only.' AS Message, 0 AS Flag;
			RETURN;
		END

		INSERT INTO Sundar_tbl_Menu (MenuName, MenuURL, is_Active, Created_On)
		VALUES (@FullName, @MenuURL, 1, ISNULL(@CreatedOn, 1));

		SELECT 'Menu added successfully.' AS Message, 1 AS Flag, SCOPE_IDENTITY() AS NewMenuId;
	END

	ELSE IF @Action = 'GetMenus'
	BEGIN
		SELECT 
			Menu_ID, MenuName, MenuURL, is_Active, is_Delete, Created_Date,
			CASE 
				WHEN @Menu_ID IS NOT NULL THEN 'SingleMenu'
				ELSE 'AllMenus'
			END AS ResultType
		FROM Sundar_tbl_Menu 
		WHERE is_Delete = 0
		  AND (@Menu_ID IS NULL OR Menu_ID = @Menu_ID)
		  AND (@IsActive IS NULL OR is_Active = @IsActive);
	END

	ELSE IF @Action = 'UpdateMenu'
	BEGIN
		IF @Role_Id != 2
		BEGIN
			SELECT 'Access Denied: Admin only.' AS Message, 0 AS Flag;
			RETURN;
		END

		UPDATE Sundar_tbl_Menu 
		SET 
			MenuName = ISNULL(@FullName, MenuName),
			MenuURL = ISNULL(@MenuURL, MenuURL),
			is_Active = ISNULL(@IsActive, is_Active),
			Update_Date = GETDATE(),
			Update_on = ISNULL(@UpdaterId, Update_on)
		WHERE Menu_ID = @Menu_ID AND is_Delete = 0;

		SELECT @@ROWCOUNT AS RowsAffected;
	END

	ELSE IF @Action = 'DeleteMenu'
	BEGIN
		IF @Role_Id != 2
		BEGIN
			SELECT 'Access Denied: Admin only.' AS Message, 0 AS Flag;
			RETURN;
		END

		-- Soft delete
		UPDATE Sundar_tbl_Menu 
		SET is_Delete = 1, Update_Date = GETDATE(), Update_on = @UpdaterId
		WHERE Menu_ID = @Menu_ID AND is_Delete = 0;

		SELECT @@ROWCOUNT AS RowsAffected;
	END

	ELSE IF @Action = 'GetLeadApplicationDetails'
	BEGIN
		DECLARE @sql NVARCHAR(MAX);
		SELECT @sql = 'SELECT * FROM Sundar_tbl_Lead WHERE Id = ' + CAST(@LeadId AS NVARCHAR(10));
		EXEC sp_executesql @sql;
	END

	-- Access Table
	ELSE IF @Action = 'GetUserAccessDetails'
    BEGIN
        SELECT 
            u.User_ID,
            u.FullName,
            u.Email,
            a.Role_ID,
            r.RoleName,
            a.Menu_ID,
            a.Access_ID
        FROM Sundar_tbl_UserDetails u
        INNER JOIN Sundar_tbl_Access a ON u.User_ID = a.User_ID AND a.is_Delete = 0
        INNER JOIN Sundar_tbl_Role r ON a.Role_ID = r.Role_ID
        WHERE u.is_Delete = 0;
    END
    ELSE IF @Action = 'GetMenuDetails'
    BEGIN
        SELECT 
            Menu_ID,
            MenuName,
            MenuURL
        FROM Sundar_tbl_Menu 
        WHERE is_Delete = 0 AND is_Active = 1;
    END
    ELSE IF @Action = 'UpdateUserAccess'
    BEGIN
        BEGIN TRY
            -- If @MenuIds is NULL or empty, use role-based defaults
            IF @MenuIds IS NULL OR LTRIM(RTRIM(@MenuIds)) = ''
            BEGIN
                SET @MenuIds = CASE 
                    WHEN @Role_Id = 2 THEN '1,2,3,5,8,9,10,12' 
                    WHEN @Role_Id = 1 THEN '1,4,6,7,11' 
                    ELSE '' 
                END;
            END

            UPDATE Sundar_tbl_Access 
            SET 
                Role_ID = @Role_Id,
                Menu_ID = @MenuIds,
                Update_Date = GETDATE(),
                Update_on = @UpdaterId
            WHERE User_ID = @UserId AND is_Delete = 0;

            SELECT 'Access updated successfully' AS Message, 1 AS Flag;
        END TRY
        BEGIN CATCH
            SELECT 'Error updating access: ' + ERROR_MESSAGE() AS Message, 0 AS Flag;
        END CATCH
    END
    ELSE IF @Action = 'GetUserAccessByUserId'
    BEGIN
        SELECT 
            a.Access_ID,
            a.User_ID,
            a.Role_ID,
            r.RoleName,
            a.Menu_ID,
            u.FullName,
            u.Email
        FROM Sundar_tbl_Access a
        INNER JOIN Sundar_tbl_UserDetails u ON a.User_ID = u.User_ID
        INNER JOIN Sundar_tbl_Role r ON a.Role_ID = r.Role_ID
        WHERE a.User_ID = @UserId AND a.is_Delete = 0 AND u.is_Delete = 0;
    END
    ELSE
    BEGIN
        SELECT 'Invalid Action' AS Message, 0 AS Flag;
    END
END
